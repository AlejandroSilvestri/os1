841 template<class TDescriptor, class F>
842 void TemplatedVocabulary<TDescriptor,F>::initiateClustersKMpp(
843  const vector<pDescriptor> &pfeatures, vector<TDescriptor> &clusters) const
844 {
845  // Implements kmeans++ seeding algorithm
846  // Algorithm:
847  // 1. Choose one center uniformly at random from among the data points.
848  // 2. For each data point x, compute D(x), the distance between x and the nearest
849  // center that has already been chosen.
850  // 3. Add one new data point as a center. Each point x is chosen with probability
851  // proportional to D(x)^2.
852  // 4. Repeat Steps 2 and 3 until k centers have been chosen.
853  // 5. Now that the initial centers have been chosen, proceed using standard k-means
854  // clustering.
855 
856  DUtils::Random::SeedRandOnce();
857 
858  clusters.resize(0);
859  clusters.reserve(m_k);
860  vector<double> min_dists(pfeatures.size(), std::numeric_limits<double>::max());
861 
862  // 1.
863 
864  int ifeature = DUtils::Random::RandomInt(0, pfeatures.size()-1);
865 
866  // create first cluster
867  clusters.push_back(*pfeatures[ifeature]);
868 
869  // compute the initial distances
870  typename vector<pDescriptor>::const_iterator fit;
871  vector<double>::iterator dit;
872  dit = min_dists.begin();
873  for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
874  {
875  *dit = F::distance(*(*fit), clusters.back());
876  }
877 
878  while((int)clusters.size() < m_k)
879  {
880  // 2.
881  dit = min_dists.begin();
882  for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
883  {
884  if(*dit > 0)
885  {
886  double dist = F::distance(*(*fit), clusters.back());
887  if(dist < *dit) *dit = dist;
888  }
889  }
890 
891  // 3.
892  double dist_sum = std::accumulate(min_dists.begin(), min_dists.end(), 0.0);
893 
894  if(dist_sum > 0)
895  {
896  double cut_d;
897  do
898  {
899  cut_d = DUtils::Random::RandomValue<double>(0, dist_sum);
900  } while(cut_d == 0.0);
901 
902  double d_up_now = 0;
903  for(dit = min_dists.begin(); dit != min_dists.end(); ++dit)
904  {
905  d_up_now += *dit;
906  if(d_up_now >= cut_d) break;
907  }
908 
909  if(dit == min_dists.end())
910  ifeature = pfeatures.size()-1;
911  else
912  ifeature = dit - min_dists.begin();
913 
914  clusters.push_back(*pfeatures[ifeature]);
915 
916  } // if dist_sum > 0
917  else
918  break;
919 
920  } // while(used_clusters < m_k)
921 
922 }

1306 void TemplatedVocabulary<TDescriptor,F>::transform(const TDescriptor &feature,
1307  WordId &word_id, WordValue &weight, NodeId *nid, int levelsup) const
1308 {
1309  // propagate the feature down the tree
1310  vector<NodeId> nodes;
1311  typename vector<NodeId>::const_iterator nit;
1312 
1313  // level at which the node must be stored in nid, if given
1314  const int nid_level = m_L - levelsup;
1315  if(nid_level <= 0 && nid != NULL) *nid = 0; // root
1316

1318  int current_level = 0;
1319 
1320  do
1321  {
1322  ++current_level;
1323  nodes = m_nodes[final_id].children;
1324  final_id = nodes[0];
1325 
1326  double best_d = F::distance(feature, m_nodes[final_id].descriptor);
1327

1329  {

1333  {
1334  best_d = d;
1335  final_id = id;
1336  }
1337  }
1338 
1339  if(nid != NULL && current_level == nid_level)
1340  *nid = final_id;
1341 
1342  } while( !m_nodes[final_id].isLeaf() );
